USE realestate
CREATE TABLE estate(
    id INT(11) NOT NULL AUTO_INCREMENT,
    type VARCHAR(20) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL,
    location VARCHAR(60) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL,
    picture VARCHAR(50) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL,
    price DOUBLE NOT NULL,
    rentOrBuy VARCHAR(10) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL,
    isActive TINYINT(1) NOT NULL,
    PRIMARY KEY (id)
);
